CREATE SCHEMA DESAFIO;
USE DESAFIO;

CREATE TABLE Alunos(
    id_aluno INT(8) PRIMARY KEY NOT NULL auto_increment,
    nome varchar(45) not null,
    periodo int(5) not null,
    curso varchar(50) not null
);

CREATE TABLE Notas(
    id_nota int(8) PRIMARY KEY NOT NULL auto_increment,
    nota float(8) NOT NULL,
    disciplina varchar(50) not null,
    id_aluno INT(8) not null,
    foreign key (id_aluno) references Alunos(id_aluno)
        ON DELETE CASCADE
        ON UPDATE CASCADE
);

FROM ALUNOS
